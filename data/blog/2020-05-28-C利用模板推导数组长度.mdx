```cpp
// 任意类型数组
template <typename T, int N>
int ArraySize1(T(&arr)[N]) {
	return N;
}

// 指定类型数组，比如SOCKET
template <int N>
int ArraySize2(SOCKET(&arr)[N]) {
	return N;
}

int main() {
	int a[10];
	std::cout << ArraySize1(a) << std::endl;		//输出结果为10
}
```
