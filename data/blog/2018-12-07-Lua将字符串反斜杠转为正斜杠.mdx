```lua
function String_conversion(value)
	local path = ""
	for i=1,#value do
        --获取当前下标字符串
		local tmp = string.sub(value,i,i)
        --如果为'\\'则替换
		if tmp=='\\' then
			path = path..'/'
		else
			path = path..tmp
		end
	end
	return path
end
```
