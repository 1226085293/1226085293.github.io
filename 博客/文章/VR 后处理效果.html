                                                          # # е®һзҺ°

                                                          # 1. жҠҠеұҸе№•еҲҶжҲҗдёӨеқ—

                                                          uv еқҗж ҮиҢғеӣҙпјҡ0-1
                                                          иҝҷйҮҢжҲ‘д»¬йңҖиҰҒеҜ№ uv иҝӣиЎҢеӨ„зҗҶпјҢжҖқи·ҜеҫҲз®ҖеҚ•пјҢжҲ‘д»¬жҠҠеҚҠеқ—еұҸе№•еҪ“дҪң uv зҡ„е®һйҷ…еқҗж ҮиҢғеӣҙпјҢд№ҹе°ұжҳҜ (0 ~ 0.5) е’Ң (0.5 ~ 1.0)

                                                          дёҖеҸҘд»Јз Ғжҗһе®ҡ

                                                          vec2 v_uv2 = mod(v_uv * 2.0, 1.0);

                                                          ж•Ҳжһңпјҡ

                                                          зңӢеҲ°дәҶиҫ“еҮәз»“жһңпјҢиҝҷеҸҘд»Јз ҒдёҚд»…жҠҠе·ҰеҸіеҲҶжҲҗдәҶдёӨиҫ№пјҢдёҠдёӢд№ҹжҳҜпјҢйӮЈд№ҲжҲ‘д»¬жҖҺд№ҲеӨ„зҗҶе‘ўпјҹ

                                                          д№ҹеҫҲз®ҖеҚ•

                                                          • uv.y еқҗж Үз§»еҠЁеҲ°дёӯй—ҙпјҢд№ҹе°ұжҳҜ +0.25
                                                          • иҝҮж»ӨжҺүеӨҡдҪҷз”»йқў
                                                              // uv еұ…дёӯ
    vec2 v_uv2 = vec2(v_uv.x, v_uv.y + 0.25);
    // uv еҲҶеқ—
    vec2 v_uv3 = mod(v_uv2 * 2.0, 1.0);
    // иҝҮж»Өз”»йқў
    if (v_uv2.y < 0.5 || v_uv2.y > 1.0 || v_uv3.x < 0.0 || v_uv3.x > 1.0 || v_uv3.y < 0.0 || v_uv3.y > 1.0) {
      discard;
    }

                                                          ж•Ҳжһңпјҡ

                                                          # 2. ж·»еҠ  CRT иҖҒз”өи§Ҷж•Ҳжһң


                                                          еҰӮжһңзәўиүІжЎҶжҳҜжҲ‘д»¬жңҖејҖе§Ӣз•ҢйқўпјҢйӮЈжҲ‘д»¬зҡ„йңҖжұӮжҳҜпјҡ

                                                          • uv.x еңЁпјҲ0 ~ 0.5пјүд»ҺеӨ§еҲ°е°ҸйҖ’еўһпјҢеңЁпјҲ0.5 ~ 1.0пјүд»Һе°ҸеҲ°еӨ§йҖ’еҮҸ
                                                          • uv.y еңЁпјҲ0 ~ 0.5пјүд»ҺеӨ§еҲ°е°ҸйҖ’еўһпјҢеңЁпјҲ0.5 ~ 1.0пјүд»Һе°ҸеҲ°еӨ§йҖ’еҮҸ
                                                            дёҠ 
                                                          vec2 post_process_crt(vec2 uv_pos_v2_, float bend_f_) {
	// -1.0 ~ 1.0
	uv_pos_v2_ = (uv_pos_v2_ - 0.5) * 2.0;

    // зј©ж”ҫ uvпјҢеўһеҠ е·®ејӮиҢғеӣҙ
	uv_pos_v2_ *= 1.5;

    // uv еҸҳеҪў
    // з”ұдәҺ uv_pos_v2_.y иҢғеӣҙеңЁпјҲ-1.0пјҢ1.0пјүпјҢжүҖд»Ҙ absпјҲuv_pos_v2_.yпјүи¶ҠеӨ§еҲҷз»“жһңи¶ҠеӨ§пјҢ0 дёәеҲҶз•Ңзәҝ
    // pow((abs(uv_pos_v2_.y) / bend_f_), 2.0) зҡ„з»“жһңз”ұдәҺ abs(uv_pos_v2_.yпјүдёҚдјҡи¶…иҝҮ 1.0пјҢжүҖд»Ҙ uv_pos_v2_.x дёҚдјҡи¶…иҝҮиҮӘе·ұ2еҖҚ
	uv_pos_v2_.x *= 1.0 + pow((abs(uv_pos_v2_.y) / bend_f_), 2.0);
	uv_pos_v2_.y *= 1.0 + pow((abs(uv_pos_v2_.x) / bend_f_), 2.0);

	// иҪ¬жҚўеӣһ 0.0 - 1.0 з©әй—ҙ
	uv_pos_v2_  = (uv_pos_v2_ / 2.0) + 0.5;

	return uv_pos_v2_;
}
